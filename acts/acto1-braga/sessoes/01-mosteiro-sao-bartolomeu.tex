\Sessao{1}{O Mosteiro de São Bartolomeu}
\Gancho{Um livro carbonizado reaparece no altar; rumores de milagres silenciosos circulam.}
\begin{Encontro}{A Vigília Quebrada}
Atmosfera de mistério e horror; sombras ecoam sussurros inaudíveis.
\Teste{Percepção + Prontidão}{7}
\end{Encontro}
\begin{Desafio}{As Três Marcas no Pergaminho}
Pistas conectam o culto aos Assamitas e Setitas nas Taifas.
\Teste{Erudição + Ocultismo}{8}
\end{Desafio}
\begin{Consequencias}
Investigações chamam atenção de aliados e antagonistas; o equilíbrio político em Braga oscila.
\end{Consequencias}
